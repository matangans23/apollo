21    // initialized in the JUnit test method first
22    public void setUp() {
23      found = new ArrayList<Integer>();
24      textArr = new char[text.length()];
25      patternArr = new char[pattern.length()];
26      for (int i = 0; i < text.length(); i++) {
27        textArr[i] = text.charAt(i);
28      }
29      for (int j = 0; j < pattern.length(); j++) {
30        patternArr[j] = pattern.charAt(j);
31      }
32      kmp = new KMP(textArr);
33      found = kmp.search(patternArr);
34    }

36    @After
37    public void tearDown() {
38      text = "";
39      pattern = "";
40      textArr = null;
41      patternArr = null;
42    }
