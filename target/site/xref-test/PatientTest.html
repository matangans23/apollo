52    @Test
53    public void visitDatumTest() {
54      String audioRecording = "audio";
55      Encryption.registerEncryption();
56      List<String> details = new ArrayList<String>();
57      details.add("id");
58      details.add("doctor");
59      details.add("visit");
60      details.add("date");
61      VisitDatum visit = new VisitDatum(details, Encryption.encrypt("transcript"), null, "time",
62          Encryption.encrypt("type"));
63      visit.setID("2");
64      assertTrue(visit.getID().equals("2"));
65      visit.setAudioRecording(Encryption.encrypt(audioRecording));
66      assertTrue(visit.getAudioRecording().equals(audioRecording));
67      visit.setDate("2/3/1994");
68      assertTrue(visit.getDate().equals("2/3/1994"));
69      visit.setTranscript(Encryption.encrypt("newTranscript"));
70      assertTrue(visit.getTranscript().equals("newTranscript"));
71      visit.setTime("5:00");
72      assertTrue(visit.getTime().equals("5:00"));
73      assertTrue(visit.getPID().equals("visit"));
74    }
