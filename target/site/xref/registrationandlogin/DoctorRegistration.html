10  /**
11   * This is our registration class that inserts into out doctor table in our
12   * database, and 'registers' a new user (doctor).
13   */
14  public class DoctorRegistration implements RegisterData {
15    /**
16     * This is a method where we take in a list of values that are information about
17     * the doctor ( email, phone, username), which will be used for registering the
18     * doctor and using that information they will be able to login.
19     *
20     * @param registrationDetails , the list of information to insert into db.
21     */
22    @Override
23    public void register(List<String> registrationDetails) {
24      // get db connection.
25      Connection conn = Database.getConn();
26      PreparedStatement prep;
27      final int seven = 7;
28      final int eight = 8;
29      final int nine = 9;
30      try {
31        prep = conn.prepareStatement("INSERT INTO doctor VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?);");
32        UUID uuid = UUID.randomUUID();
33        prep.setString(1, uuid.toString()); // UUID
34        prep.setString(2, registrationDetails.get(0)); // first name
35        prep.setString(3, registrationDetails.get(1)); // middle name
36        prep.setString(4, registrationDetails.get(2)); // last name
37        prep.setString(5, registrationDetails.get(3)); // email
38        prep.setString(6, registrationDetails.get(4)); // username
39        prep.setBytes(seven, Encryption.encrypt(registrationDetails.get(5))); // password
40        prep.setString(eight, registrationDetails.get(6)); // phone number
41        prep.setString(nine, registrationDetails.get(seven)); // instituion
42        prep.addBatch();
43        prep.executeBatch();
44      } catch (Exception e) {
45        System.err.println("ERROR: SQL error inserting into Db");
46        return;
47      }
48  
49    }
50  }
