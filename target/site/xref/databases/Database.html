40    /**
41     * Instantiates the database, creating tables if necessary. Automatically loads
42     * files.
43     *
44     * @return int , 1 for correct, 0 for errors.
45     */
46    public static int makeDatabase() {
47      // Initializes the database connection, turns foreign keys on.
48      String filename = "data/db/apollo.sqlite3";
49      try {
50        FileReader file = new FileReader(filename);
51      } catch (FileNotFoundException e1) {
52        System.err.println("ERROR: File does not exist");
53        return 0;
54      }
55      try {
56        Class.forName("org.sqlite.JDBC");
57      } catch (ClassNotFoundException e) {
58        System.err.println("ERROR: Database");
59        return 0;
60      }
61      String urlToDB = "jdbc:sqlite:" + filename;
62      try {
63        conn = DriverManager.getConnection(urlToDB);
64      } catch (SQLException e) {
65        System.err.println("ERROR: Database");
66        return 0;
67      }
68      // these two lines tell the database to enforce foreign keys during
69      // operations,
70      // and should be present
71      Statement stat;
72      try {
73        stat = conn.createStatement();
74        stat.executeUpdate("PRAGMA foreign_keys=ON;");
75      } catch (SQLException e) {
76        System.err.println("ERROR: Database");
77        return 0;
78      }
79      return 1;
80    }
