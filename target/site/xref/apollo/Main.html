450     /**
451      * This is a parser that is used to convert dates from the JavaScript format
452      * into the format that we use when registering a visit. This is needed for us
453      * to properly search for visits between date ranges.
454      *
455      * @param date A String, representing a date to convert into our database's
456      *             format.
457      * @return A String, the same date into tehhe correct format: "day-month-year".
458      */
459     private String dateProcessor(String date) {
460       String[] dateArray = new String[3];
461       String[] splitDate = date.split("\\s+");
462       dateArray[0] = splitDate[1]; // month
463       dateArray[1] = splitDate[2]; // day
464       dateArray[2] = splitDate[3]; // year
465       Map<String, String> months = new HashMap<String, String>();
466       months.put("Jan", "01");
467       months.put("Feb", "02");
468       months.put("Mar", "03");
469       months.put("Apr", "04");
470       months.put("May", "05");
471       months.put("Jun", "06");
472       months.put("Jul", "07");
473       months.put("Aug", "08");
474       months.put("Sep", "09");
475       months.put("Oct", "10");
476       months.put("Nov", "11");
477       months.put("Dec", "12");
478       String numericMonth = months.get(dateArray[0]);
479       dateArray[0] = numericMonth;
480       StringBuilder toRet = new StringBuilder();
481       toRet.append(dateArray[2]);
482       toRet.append("-");
483       toRet.append(dateArray[0]);
484       toRet.append("-");
485       toRet.append(dateArray[1]);
486       return toRet.toString();
487     }
